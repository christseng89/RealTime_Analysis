# Install mysql
sudo apt install mysql-server

#Start mysql
sudo systemctl start mysql.service

#Start the client
sudo mysql

#Create database 
create database analytics;

#use analytics DB
use analytics;

#Grant access in mysql
CREATE USER 'analyticsuser'@'[::1]' IDENTIFIED BY 'password';
GRANT ALL PRIVILEGES ON analytics.* TO 'analyticsuser'@'[::1]';

#Create mysql location table
CREATE TABLE Location (
    location_id varchar(255) NOT NULL,
    city varchar(255),
    country varchar(255),
    PRIMARY KEY (location_id)
);
